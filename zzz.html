text.z <- read_excel("수자원키워드.xlsx")

text.v_df <-  text.v %>% 
  unnest_tokens(word, text, token = "words") %>% 
  inner_join(senti_dic_df) %>% 
  count(word, sScore, sort = T) %>% 
  filter(str_length(word) > 1) %>% 
  mutate(word = reorder(word, n)) %>% 
  slice_head(n = 20)

text.v_df %>%

text.z_df <- text.z %>%
